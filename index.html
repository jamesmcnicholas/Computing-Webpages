FizzBuzz 0-100 Written in Java 
import java.util.Scanner;
public class FizzBuzz {
public static void main(String[] args){
	Scanner input = new Scanner(System.in);
	System.out.println("Enter the 'Fizz' number");
	int fizz = input.nextInt();
	System.out.println("Enter the 'Buzz' number");
	int buzz=input.nextInt();
	for (int i=0;i<=100;i++){
		String out="";
		if (i%fizz==0){
			out+="Fizz";
		}
		if (i%buzz==0){
			out+="Buzz";
		}
		if (out.equals("")) {
			System.out.println(i);
		}else{
			System.out.println(out);
		}
	}
}
}

Calculator Crafted with Java

An object oriented Calculator class 

public class Calculator {
    private int value1;
    private int value2;
    private Reader reader = new Reader();

    public void start() {
        while (true) {
            System.out.print("command: ");
            String command = reader.readString();
            if (command.equals("end")) {
                break;
            } else if (command.equals("sum")) {
                sum();
            } else if (command.equals("difference")) {
                difference();
            } else if (command.equals("product")) {
                product();
            }
        }
    }
    private void sum(){
        readValues();
        int result = value1+ value2;
        System.out.println("sum of the values "+ result);
    }
	private void readValues(){
		System.out.println("value1:");
		this.value1= reader.readInteger();
		System.out.println("value2:");
		this.value2= reader.readInteger();
	}
}

Guesser! Written in Java
A basic number guessing game using iteration and methods. 
import java.util.Random;
import java.util.Scanner;
public class GuessingNumberGame {
    public static void main(String[] args) {
        Scanner reader = new Scanner(System.in);
        int numberDrawn = drawNumber();
        int guesses=0;
        int guess=0;
        while (true) {
            System.out.println("Type a number");
            guess = Integer.parseInt(reader.nextLine());
            guesses++;
            if (guess > numberDrawn) {
                System.out.println("Too high");
            }else if (guess == numberDrawn) {
                System.out.println("Correct!");
                System.out.println("You used "+guesses+" guesses.");
                break;
            }else{ System.out.println("Too low");}
        }
    }
    private static int drawNumber() {
			return new Random().nextInt(101);
		}
}
